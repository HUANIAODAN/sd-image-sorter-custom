    // 计算位置并设置
    card.style.left = shortestCol × (列宽 + 间距) + 'px';
    card.style.top = columnHeights[shortestCol] + 'px';
    
    // 更新列高度
    columnHeights[shortestCol] += card.height + 间距;
  });
  
  // 4. 设置容器高度
  container.style.height = Math.max(...columnHeights) + 'px';
}
